
    getContext('2d')

	•	کاربرد: این متد برای گرفتن یک محتوای ۲بعدی روی بوم استفاده می‌شود که امکان رسم و طراحی‌های گرافیکی را فراهم می‌کند.
	مثال:const canvas = document.getElementById('myCanvas');
    const ctx = canvas.getContext('2d');

    fillRect(x, y, width, height)

	•	کاربرد: رسم یک مستطیل توپر (filled rectangle) در موقعیت (x, y) با عرض و ارتفاع مشخص.
	•	مثال:ctx.fillStyle = "blue";
    ctx.fillRect(20, 20, 100, 50); // رسم مستطیل

    •strokeRect(x, y, width, height)

	کاربرد: رسم یک مستطیل با خطوط حاشیه‌ای (خطوط مرزی) در موقعیت (x, y) با عرض و ارتفاع مشخص.
    مثال:
    ctx.strokeStyle = "green";
    ctx.strokeRect(150, 20, 100, 50); // رسم مستطیل با خط مرزی

    clearRect(x, y, width, height)

	•	کاربرد: پاک کردن یک ناحیه از بوم با ابعاد مشخص. این متد برای حذف قسمتی از بوم استفاده می‌شود.
	•	مثال:ctx.clearRect(20, 20, 100, 50); // پاک کردن مستطیل از بوم

    •beginPath()

		کاربرد: شروع یک مسیر جدید برای رسم. این متد قبل از رسم هر شکلی (مانند خط یا دایره) باید فراخوانی شود.
	•	مثال:ctx.beginPath(); // شروع مسیر جدید

    moveTo(x, y)

	•	کاربرد: حرکت به مختصات (x, y) بدون رسم یک خط. این متد برای شروع مسیر خطی از یک نقطه به نقطه دیگر استفاده می‌شود.
	•	مثال:ctx.moveTo(50, 50); // حرکت به نقطه (50, 50)

    lineTo(x, y)

	•	کاربرد: رسم یک خط از موقعیت جاری به مختصات (x, y).
	•	مثال:ctx.lineTo(150, 50); // رسم خط از (50, 50) به (150, 50)

    stroke()

	•	کاربرد: رسم خطوط مسیر تعریف شده توسط beginPath(), moveTo(), و lineTo() یا دیگر توابع مسیر.
	•	مثال:ctx.stroke(); // رسم مسیر

    arc(x, y, radius, startAngle, endAngle, anticlockwise)

	•	کاربرد: رسم یک دایره یا بخشی از دایره (کمان) با مرکز (x, y) و شعاع مشخص.
	•	مثال:ctx.arc(100, 100, 50, 0, Math.PI * 2); // رسم دایره کامل

    fill()

	•	کاربرد: پر کردن داخل مسیر با رنگی که در fillStyle تنظیم شده است.
	•	مثال:ctx.fillStyle = "red";
ctx.fill(); // پر کردن شکل با رنگ قرمز

    strokeStyle

	•	کاربرد: تنظیم رنگ خط مرزی (stroke) برای رسم خطوط.
	•	مثال:ctx.strokeStyle = "black"; // تنظیم رنگ خط به سیاه

    lineWidth

	•	کاربرد: تنظیم ضخامت خط برای خطوط مرزی (stroke).
	•	مثال:ctx.lineWidth = 5; // تنظیم ضخامت خط به 5 پیکسل

    fillStyle

	•	کاربرد: تنظیم رنگ داخلی شکل‌ها (مانند مستطیل‌ها، دایره‌ها).
	•	مثال:ctx.fillStyle = "blue"; // تنظیم رنگ داخل به آبی

    setLineDash([length, gap])

	•	کاربرد: تنظیم خط چین یا الگوی خط برای رسم خطوط. length طول هر خط و gap فاصله بین خطوط را تعیین می‌کند.
	•	مثال:ctx.setLineDash([5, 3]); // رسم خطوط چین‌دار

    drawImage(image, x, y, width, height)

	•	کاربرد: رسم یک تصویر روی بوم در مختصات (x, y) با ابعاد مشخص.
	•	مثال:const img = new Image();
img.src = 'image.jpg';
img.onload = () => {
  ctx.drawImage(img, 0, 0, 200, 200); // رسم تصویر در موقعیت (0, 0)
};

    save() و restore()

	•	کاربرد: ذخیره و بازیابی وضعیت فعلی context (شامل رنگ‌ها، مسیرها و تغییرات دیگر). این متد برای ایجاد تغییرات موقتی در context مفید است.
	•	مثال:ctx.save(); // ذخیره وضعیت
ctx.fillStyle = "green"; // تغییر رنگ
ctx.fillRect(50, 50, 100, 100); // رسم مستطیل با رنگ جدید
ctx.restore(); // بازگرداندن وضعیت قبلی

    rotate(angle)

	•	کاربرد: چرخاندن همه اشکال و مسیرهای بعدی در بوم به اندازه زاویه داده شده (بر حسب رادیان).
	•	مثال:ctx.rotate(Math.PI / 4); // چرخاندن به اندازه 45 درجه
ctx.fillRect(0, 0, 100, 100); // رسم مستطیل بعد از چرخاندن

    translate(x, y)

	•	کاربرد: انتقال (جابجایی) همه اشکال و مسیرهای بعدی در بوم به مختصات جدید.
	•	مثال:ctx.translate(50, 50); // جابجایی به موقعیت (50, 50)
ctx.fillRect(0, 0, 100, 100); // رسم مستطیل در موقعیت جدید
